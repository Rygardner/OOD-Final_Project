408    def horizontal_movement(self, keys: pygame.key.ScancodeWrapper) -> int:
409        """Handles Horizontal (player.x) movement
410
411        Args
412            keys (pygame.key.ScancodeWrapper): Keyboard input
413        """
414        return self.movement_strategy.get_horizontal_velocity(self.move_speed, keys)
415
416    def handle_horizontal_collisions(
417            self, colliders: list[pygame.Rect], horizontal_velocity: int,) -> None:
418        """Handles horzontal collisions
419
420        Args
421            colliders (list[pygame.Rect]): List of walls or anything the player
422                                           should collide horizontally with
423        """
424        self.touching_left_wall = False
425        self.touching_right_wall = False
426
427        for collider in colliders:
428            if self.rect.colliderect(collider):
429                if horizontal_velocity > 0:
430                    self.rect.right = collider.left
431                    self.touching_right_wall = True
432                elif horizontal_velocity < 0:
433                    self.rect.left = collider.right
434                    self.touching_left_wall = True
435
436    def jump(self, keys: pygame.key.ScancodeWrapper) -> None:
437        """Handles jump logic
438        """
439        if keys[pygame.K_SPACE]:
440            if self.on_ground:
441                self.jump_velocity = -self.jump_speed
442                self.on_ground = False
443                self.can_wall_jump = True
444            elif self.can_wall_jump and (self.touching_left_wall or self.touching_right_wall):
445                self.jump_velocity = -self.wall_jump_speed
446                self.can_wall_jump = False
447
448    def apply_gravity(self) -> None:
449        """Applies gravity to player
450        """
451        self.jump_velocity += self.fall_speed
452        self.y += int(self.jump_velocity)
453
454    def handle_vertical_collision(self, colliders: list[pygame.Rect]) -> None:
455        """Handles vertical collision. So capable of landing on a floor/platform. Also
456        handles collision for when the player jumps and collides with a platform from
457        underneath.
458
459        Args:
460            colliders (list[pygame.Rect]): List of platforms, floor, or anything the player should
461                                           vertically collide with.
462        """
463        self.on_ground = False
464
465        for collider in colliders:
466            if self.rect.colliderect(collider):
467                if self.jump_velocity > 0:
468                    self.rect.bottom = collider.top
469                    self.jump_velocity = 0.0
470                    self.on_ground = True
471                    self.can_wall_jump = True
472                elif self.jump_velocity < 0:
473                    self.rect.top = collider.bottom
474                    self.jump_velocity = 0.0
475
476    def reposition(self, x: int, y: int) -> None:
477        """Move the player to a new (x, y) area. Could use for spawning/respawning
478        or teleport the player to set location. Resets jump_velocity and flags
479
480        Args:
481            x (int): player's new x position
482            y (int): player's new y position
483        """
484        if not isinstance(x, int) or not isinstance(y, int):
485            raise TypeError("spawn coordinates must be ints")
486
487        self.rect.x = x
488        self.rect.y = y
489
490        self.jump_velocity = 0.0
491        self.on_ground = False
492        self.touching_left_wall = False
493        self.touching_right_wall = False
494        self.can_wall_jump = True
495
496    def update(self, keys: pygame.key.ScancodeWrapper, colliders: list[pygame.Rect]) -> None:
497        """Updates player's loop in game
498
499        Args
500            keys (pygame.key.ScancodeWrapper): Keyboard input
501            colliders (list[pygame.Rect]): List of wall, platforms, or anything the player can
502                                           collide with
503        """
504        horizontal_velocity = self.horizontal_movement(keys)
505        self.x += horizontal_velocity
506
507        self.handle_horizontal_collisions(colliders, horizontal_velocity)
508
509        self.jump(keys)
510
511        self.apply_gravity()
512
513        self.handle_vertical_collision(colliders)
514
515    def draw(self, surface: pygame.Surface) -> None:
516        """Draw player onto pygame screen/surface
517
518        Args:
519            surface (pygame.Surface): The window/surface to draw the player on
520        """
521        pygame.draw.rect(surface, self.color, self.rect)
